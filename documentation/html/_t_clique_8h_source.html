41 void set_from_product(const TMatrix &First, const TMatrix &Second) {
42 // do matrix multiplication
43 // _mat(0,0) = First(0,0) * Second(0,0) + First(0,1) * Second(1,0)
44 _mat[0] = First[0] * Second[0] + First[2] * Second[1];
45 _mat[2] = 1.0 - _mat[0];
46 _mat[1] = First[1] * Second[0] + First[3] * Second[1];
47 _mat[3] = 1.0 - _mat[1];

105 // calculate matrix exponential for first bin

107 P_0.set_from_matrix_exponential(_lambda_c * a);

109 // calculate matrix exponential of scaling matrix
110 TMatrix _matrix_alpha;
111 _matrix_alpha.set_from_matrix_exponential(_lambda_c * Delta);

113 _matrices[0] = P_0;
114 // do recursion
115 for (size_t k = 1; k < _matrices.size(); ++k) {
116 _matrices[k].set_from_product(_matrices[k - 1], _matrix_alpha);
