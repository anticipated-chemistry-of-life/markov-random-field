24 [[nodiscard]] std::pair<bool, size_t> _binary_search(uint64_t coordinate) const {
25
26 // lower_bound return the first element that is not less than the value
27 auto it = std::lower_bound(_vec.begin(), _vec.end(), coordinate);
28
29 // if our coordinate is bigger than the biggest element in the vector
30 // we say that we haven't found our element and that if we want to
31 // insert it, we should insert it at the end of the vector
32 if (it == _vec.end()) { return {false, _vec.size()}; }
33
34 // else our coordinate is in the range of the coordinates in the vector
35 // meaning that if we haven't found it, we will insert it at that position
36 // to keep the vector sorted
37 size_t index = std::distance(_vec.begin(), it);
38 if (*it != coordinate) { return {false, index}; }
39
40 // if we found the coordinate we return the index and true
41 return {true, index};
42 };

44public:
45 explicit TStorageYVector(const size_t n_iterations, const std::vector<size_t> &dimensions) {
46 constexpr uint16_t max_value = std::numeric_limits<uint16_t>::max();
47 _thinning_factor = std::ceil(static_cast<double>(n_iterations) / static_cast<double>(max_value));
48 _total_counts = n_iterations / _thinning_factor;
49 _dimensions = dimensions;
50 };

58 [[nodiscard]] bool is_one(const uint64_t coordinate) const {
59 auto [found, index] = _binary_search(coordinate);
60 if (found) { return _vec[index].is_one(); }
61 return false;
62 };

64 bool is_one(const std::vector<size_t> &multi_dim_index) const { return is_one(get_coordinate(multi_dim_index)); }
65
66 /* set_to_one will set the element at the coordinate to 1
67 * if the element is already in the vector, we just set it to 1
68 * if the element is not in the vector, we insert it in the vector
69 * and set it to 1
70 */
71 void set_to_one(uint64_t coordinate) {
72 auto [found, index] = _binary_search(coordinate);
73 if (found) {
74 _vec[index].set_state(true);
75 } else {
76 _vec.insert(_vec.begin() + index, TStorageY(coordinate));
77 }
78 }

80 void set_to_zero(uint64_t coordinate) {
81 auto [found, index] = _binary_search(coordinate);
82 if (found) { _vec[index].set_state(false); }
83 }

85 void add_to_counter(size_t iteration) {

91 double get_fraction_of_ones(uint64_t coordinate) const {
92 auto [found, index] = _binary_search(coordinate);
93 if (!found) { return 0.0; }
94 return static_cast<double>(_vec[index].get_counter()) / static_cast<double>(_total_counts);
95 }

103 }

107 void remove_zeros() {
108 _vec.erase(std::remove_if(_vec.begin(), _vec.end(), [](const TStorageY &elem) { return !elem.is_one(); }),
109 _vec.end());
110 }
